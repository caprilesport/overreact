193    def _yield_scheme(self):
194        """Produce a renderables describing the reaction scheme.
195
196        This is meant to be used from within `__rich_console__`.
197
198        Yields
199        ------
200        renderable
201        """
202        scheme = rx.core._check_scheme(self.model.scheme)
203
204        raw_table = Table(
205            title="(read) reactions", box=self.box_style, show_header=False
206        )
207        raw_table.add_column(justify="left")
208        for r in rx.core.unparse_reactions(scheme).split("\n"):
209            raw_table.add_row(r)
210        yield Panel(raw_table, expand=False)
211
212        transition_states = rx.core.get_transition_states(
213            scheme.A, scheme.B, scheme.is_half_equilibrium
214        )
215
216        parsed_table = Table(
217            Column("no", justify="right"),
218            Column("reactant(s)", justify="left"),
219            Column("via‡", justify="left"),
220            Column("product(s)", justify="left"),
221            Column("half equilib.?", justify="center"),
222            title="(parsed) reactions",
223            box=self.box_style,
224        )
225        for i, reaction in enumerate(scheme.reactions):
226            reactants, _, products = re.split(r"\s*(->|<=>|<-)\s*", reaction)
227            row = [f"{i:d}", reactants, None, products, "No"]
228            if transition_states[i] is not None:
229                row[2] = scheme.compounds[transition_states[i]]
230            elif scheme.is_half_equilibrium[i]:
231                row[4] = "Yes"
232            parsed_table.add_row(*row)
233        yield parsed_table

235    def _yield_compounds(self):
236        """Produce a renderables describing the compounds.
237
238        This is meant to be used from within `__rich_console__`.
239
240        Yields
241        ------
242        renderable
243
244        Raises
245        ------
246        ValueError
247            If at least one compound has undefined data.
248        """
249        undefined_compounds = []
250        for name in self.model.compounds:
251            if not self.model.compounds[name]:
252                undefined_compounds.append(name)
253        if undefined_compounds:
254            raise ValueError(f"undefined compounds: {', '.join(undefined_compounds)}")
255
256        logfiles_table = Table(
257            Column("no", justify="right"),
258            Column("compound", justify="left"),
259            Column("path", justify="left"),
260            title="logfiles",
261            box=self.box_style,
262        )
263        compounds_table = Table(
264            Column("no", justify="right"),
265            Column("compound", justify="left"),
266            Column("elec. energy\n〈Eₕ〉", justify="center"),
267            Column("spin mult.", justify="center"),
268            Column("smallest vibfreqs\n〈cm⁻¹〉", justify="center"),
269            Column("point group", justify="center"),
270            title="compounds",
271            box=self.box_style,
272        )
273        for i, (name, data) in enumerate(self.model.compounds.items()):
274            path_text = None
275            if data.logfile is not None:
276                path_text = Text(data.logfile)
277                path_text.highlight_regex(r"[^\/]+$", "bright_blue")
278            logfiles_table.add_row(f"{i:d}", name, path_text)
279
280            vibfreqs_text = None
281            if data.vibfreqs is not None:
282                vibfreqs_text = Text(
283                    ", ".join([f"{vibfreq:+7.1f}" for vibfreq in data.vibfreqs[:3]])
284                )
285                vibfreqs_text.highlight_regex(r"-\d+\.\d", "bright_yellow")
286
287            point_group = coords.find_point_group(
288                atommasses=data.atommasses, atomcoords=data.atomcoords
289            )
290            compounds_table.add_row(
291                f"{i:d}",
292                name,
293                f"{data.energy / (constants.hartree * constants.N_A):17.12f}",
294                f"{data.mult}",
295                vibfreqs_text,
296                point_group,
297            )
298        yield logfiles_table
299        yield compounds_table

301    def _yield_thermochemistry(self):
302        """Produce a renderables describing the thermochemistry of the reaction scheme.
303
304        This is meant to be used from within `__rich_console__`.
305
306        Yields
307        ------
308        renderable
309        """
310        scheme = rx.core._check_scheme(self.model.scheme)
311
312        molecular_masses = np.array(
313            [np.sum(data.atommasses) for name, data in self.model.compounds.items()]
314        )
315        energies = np.array(
316            [data.energy for name, data in self.model.compounds.items()]
317        )
318        internal_energies = rx.get_internal_energies(
319            self.model.compounds, qrrho=self.qrrho, temperature=self.temperature
320        )
321        enthalpies = rx.get_enthalpies(
322            self.model.compounds, qrrho=self.qrrho, temperature=self.temperature
323        )
324        entropies = rx.get_entropies(
325            self.model.compounds, qrrho=self.qrrho, temperature=self.temperature
326        )
327        freeenergies = enthalpies - self.temperature * entropies
328        assert np.allclose(
329            freeenergies,
330            rx.get_freeenergies(
331                self.model.compounds,
332                qrrho=self.qrrho,
333                temperature=self.temperature,
334                pressure=self.pressure,
335            ),
336        ), "free energies do not match enthalpies and entropies"
337
338        compounds_table = Table(
339            Column("no", justify="right"),
340            Column("compound", justify="left"),
341            Column("mass\n〈amu〉", justify="center"),
342            Column("Gᶜᵒʳʳ\n〈kcal/mol〉", justify="center", style="bright_green"),
343            Column("Uᶜᵒʳʳ\n〈kcal/mol〉", justify="center"),
344            Column("Hᶜᵒʳʳ\n〈kcal/mol〉", justify="center"),
345            Column("S\n〈cal/mol·K〉", justify="center"),
346            title="estimated thermochemistry (compounds)",
347            box=self.box_style,
348        )
349        for i, (name, data) in enumerate(self.model.compounds.items()):
350            compounds_table.add_row(
351                f"{i:d}",
352                name,
353                f"{molecular_masses[i]:6.2f}",
354                f"{(freeenergies[i] - data.energy) / constants.kcal:14.2f}",
355                f"{(internal_energies[i] - data.energy) / constants.kcal:14.2f}",
356                f"{(enthalpies[i] - data.energy) / constants.kcal:14.2f}",
357                f"{entropies[i] / constants.calorie:10.2f}",
358            )
359        yield compounds_table
360
361        delta_mass = rx.get_delta(scheme.A, molecular_masses)
362        delta_energies = rx.get_delta(scheme.A, energies)
363        delta_internal_energies = rx.get_delta(scheme.A, internal_energies)
364        delta_enthalpies = rx.get_delta(scheme.A, enthalpies)
365        # TODO(schneiderfelipe): log the contribution of reaction symmetry
366        delta_entropies = rx.get_delta(scheme.A, entropies) + rx.get_reaction_entropies(
367            scheme.A, temperature=self.temperature, pressure=self.pressure
368        )
369        delta_freeenergies = delta_enthalpies - self.temperature * delta_entropies
370        assert np.allclose(
371            delta_freeenergies,
372            rx.get_delta(scheme.A, freeenergies)
373            - self.temperature
374            * rx.get_reaction_entropies(
375                scheme.A, temperature=self.temperature, pressure=self.pressure
376            ),
377        ), "reaction free energies do not match reaction enthalpies and reaction entropies"
378
379        delta_activation_mass = rx.get_delta(scheme.B, molecular_masses)
380        delta_activation_energies = rx.get_delta(scheme.B, energies)
381        delta_activation_internal_energies = rx.get_delta(scheme.B, internal_energies)
382        delta_activation_enthalpies = rx.get_delta(scheme.B, enthalpies)
383        # TODO(schneiderfelipe): log the contribution of reaction symmetry
384        delta_activation_entropies = rx.get_delta(
385            scheme.B, entropies
386        ) + rx.get_reaction_entropies(
387            scheme.B, temperature=self.temperature, pressure=self.pressure
388        )
389        delta_activation_freeenergies = (
390            delta_activation_enthalpies - self.temperature * delta_activation_entropies
391        )
392        assert np.allclose(
393            delta_activation_freeenergies,
394            rx.get_delta(scheme.B, freeenergies)
395            - self.temperature
396            * rx.get_reaction_entropies(
397                scheme.B, temperature=self.temperature, pressure=self.pressure
398            ),
399        ), "activation free energies do not match activation enthalpies and activation entropies"
400
401        circ_table = Table(
402            Column("no", justify="right"),
403            Column("reaction", justify="left"),
404            Column("Δmass°\n〈amu〉", justify="center"),
405            Column("ΔG°\n〈kcal/mol〉", justify="center", style="bright_green"),
406            Column("ΔE°\n〈kcal/mol〉", justify="center"),
407            Column("ΔU°\n〈kcal/mol〉", justify="center"),
408            Column("ΔH°\n〈kcal/mol〉", justify="center"),
409            Column("ΔS°\n〈cal/mol·K〉", justify="center"),
410            title="estimated (reaction°) thermochemistry",
411            box=self.box_style,
412        )
413        dagger_table = Table(
414            Column("no", justify="right"),
415            Column("reaction", justify="left"),
416            Column("Δmass‡\n〈amu〉", justify="center"),
417            Column("ΔG‡\n〈kcal/mol〉", justify="center", style="bright_green"),
418            Column("ΔE‡\n〈kcal/mol〉", justify="center"),
419            Column("ΔU‡\n〈kcal/mol〉", justify="center"),
420            Column("ΔH‡\n〈kcal/mol〉", justify="center"),
421            Column("ΔS‡\n〈cal/mol·K〉", justify="center"),
422            title="estimated (activation‡) thermochemistry",
423            box=self.box_style,
424        )
425        for i, reaction in enumerate(scheme.reactions):
426            if scheme.is_half_equilibrium[i]:
427                circ_row = [
428                    f"{i:d}",
429                    reaction,
430                    f"{delta_mass[i]:6.2f}",
431                    f"{delta_freeenergies[i] / constants.kcal:10.2f}",
432                    f"{delta_energies[i] / constants.kcal:10.2f}",
433                    f"{delta_internal_energies[i] / constants.kcal:10.2f}",
434                    f"{delta_enthalpies[i] / constants.kcal:10.2f}",
435                    f"{delta_entropies[i] / constants.calorie:11.2f}",
436                ]
437                dagger_row = [
438                    f"{i:d}",
439                    reaction,
440                    None,
441                    None,
442                    None,
443                    None,
444                    None,
445                    None,
446                ]
447            else:
448                circ_row = [
449                    f"{i:d}",
450                    reaction,
451                    f"{delta_mass[i]:6.2f}",
452                    f"{delta_freeenergies[i] / constants.kcal:10.2f}",
453                    f"{delta_energies[i] / constants.kcal:10.2f}",
454                    f"{delta_internal_energies[i] / constants.kcal:10.2f}",
455                    f"{delta_enthalpies[i] / constants.kcal:10.2f}",
456                    f"{delta_entropies[i] / constants.calorie:11.2f}",
457                ]
458                dagger_row = [
459                    f"{i:d}",
460                    reaction,
461                    f"{delta_activation_mass[i]:6.2f}",
462                    f"{delta_activation_freeenergies[i] / constants.kcal:10.2f}",
463                    f"{delta_activation_energies[i] / constants.kcal:10.2f}",
464                    f"{delta_activation_internal_energies[i] / constants.kcal:10.2f}",
465                    f"{delta_activation_enthalpies[i] / constants.kcal:10.2f}",
466                    f"{delta_activation_entropies[i] / constants.calorie:11.2f}",
467                ]
468
469            circ_table.add_row(*circ_row)
470            dagger_table.add_row(*dagger_row)
471        yield circ_table
472        yield dagger_table

474    def _yield_kinetics(self):
475        """Produce a renderables describing the kinetics of the system.
476
477        This is meant to be used from within `__rich_console__`.
478
479        Yields
480        ------
481        renderable
482        """
483        if isinstance(self.bias, str):
484            data = np.genfromtxt(self.bias, names=True, delimiter=",")
485            data = {name: data[name] for name in data.dtype.names}
486
487            scheme, _, y0 = _prepare_simulation(
488                self.model.scheme,
489                rx.get_k(
490                    self.model.scheme,
491                    self.model.compounds,
492                    bias=0.0,
493                    tunneling=self.tunneling,
494                    qrrho=self.qrrho,
495                    scale="l mol-1 s-1",
496                    temperature=self.temperature,
497                    pressure=self.pressure,
498                ),
499                self.concentrations,
500            )
501            self.bias = rx.get_bias(
502                scheme,
503                self.model.compounds,
504                data,
505                y0,
506                tunneling=self.tunneling,
507                qrrho=self.qrrho,
508                temperature=self.temperature,
509                pressure=self.pressure,
510                method=self.method,
511                rtol=self.rtol,
512                atol=self.atol,
513            )
514
515        # TODO(schneiderfelipe): apply other corrections to k (such as
516        # diffusion control).
517        # TODO(schneiderfelipe): use pressure.
518        k = {
519            "M⁻ⁿ⁺¹·s⁻¹": rx.get_k(
520                self.model.scheme,
521                self.model.compounds,
522                bias=self.bias,
523                tunneling=self.tunneling,
524                qrrho=self.qrrho,
525                scale="l mol-1 s-1",
526                temperature=self.temperature,
527                pressure=self.pressure,
528            ),
529            "(cm³/particle)ⁿ⁻¹·s⁻¹": rx.get_k(
530                self.model.scheme,
531                self.model.compounds,
532                bias=self.bias,
533                tunneling=self.tunneling,
534                qrrho=self.qrrho,
535                scale="cm3 particle-1 s-1",
536                temperature=self.temperature,
537                pressure=self.pressure,
538            ),
539            "atm⁻ⁿ⁺¹·s⁻¹": rx.get_k(
540                self.model.scheme,
541                self.model.compounds,
542                bias=self.bias,
543                tunneling=self.tunneling,
544                qrrho=self.qrrho,
545                scale="atm-1 s-1",
546                temperature=self.temperature,
547                pressure=self.pressure,
548            ),
549        }
550        kappa = rx.get_kappa(
551            self.model.scheme,
552            self.model.compounds,
553            method=self.tunneling,
554            qrrho=self.qrrho,
555            temperature=self.temperature,
556        )
557
558        kinetics_table = Table(
559            *(
560                [
561                    Column("no", justify="right"),
562                    Column("reaction", justify="left"),
563                    Column("half equilib.?", justify="center"),
564                ]
565                + [Column(f"k\n〈{scale}〉", justify="center") for scale in k]
566                + [Column("κ", justify="center")]
567            ),
568            title="estimated reaction rate constants",
569            box=self.box_style,
570        )
571        for i, reaction in enumerate(self.model.scheme.reactions):
572            row = (
573                [f"{i:d}", reaction, "No"]
574                + [f"{k[scale][i]:.3g}" for scale in k]
575                + [f"{kappa[i]:.3g}"]
576            )
577            if self.model.scheme.is_half_equilibrium[i]:
578                row[2] = "Yes"
579                row[-1] = None  # hide transmission coefficient
580
581            kinetics_table.add_row(*row)
582        yield kinetics_table
583        yield Markdown(
584            "Only in the table above, all Gibbs free energies were biased by "
585            f"{self.bias} J/mol."
586        )
587        yield Markdown(
588            "For **half-equilibria**, only ratios make sense: in simulations, **equilibria will be adjusted to be faster than all other reactions**."
589        )
590
591        if self.concentrations is not None and self.concentrations:
592            scheme, k, y0 = _prepare_simulation(
593                self.model.scheme, k["M⁻ⁿ⁺¹·s⁻¹"], self.concentrations
594            )
595
596            # TODO(schneiderfelipe): encapsulate everything in a function that depends
597            # on the freeenergies as first parameter
598            dydt = rx.get_dydt(scheme, k)
599
600            y, r = rx.get_y(
601                dydt,
602                y0=y0,
603                method=self.method,
604                rtol=self.rtol,
605                atol=self.atol,
606                max_time=self.max_time,
607            )
608            conc_table = Table(
609                Column("no", justify="right"),
610                Column("compound", justify="left"),
611                Column(f"t = {y.t_min:.1g} s", justify="right"),
612                Column(
613                    f"t = {y.t_max:.1g} s",
614                    justify="right",
615                    style="bright_green",
616                ),
617                title="initial and final concentrations\n〈M〉",
618                box=self.box_style,
619            )
620            for i, name in enumerate(scheme.compounds):

623                    name,
624                    f"{y(y.t_min)[i]:.3f}",
625                    f"{y(y.t_max)[i]:.3f}",
626                )

628
629            t_span = y.t_max - y.t_min
630            active = ~np.isclose(
631                y(y.t_min + 0.01 * t_span * np.random.rand()),
632                y(y.t_max - 0.01 * t_span * np.random.rand()),
633                rtol=0.01,
634            )
635            if self.plot == "all" or not np.any(active):
636                active = np.array([True for _ in scheme.compounds])
637
638            factor = y(y.t_max)[active].max()
639            reference = y(y.t_max)[active] / factor
640
641            step = 0.9  # multiply by this factor to decrease t_max
642            n_max = np.log(1e-8) / np.log(step)
643
644            # We plot until concentrations are within this value from the
645            # final concentrations.
646            alpha = 0.01
647
648            t_max, i = y.t_max, 0
649            while i < n_max and np.allclose(
650                y(t_max)[active] / factor, reference, atol=alpha
651            ):
652                t_max = step * t_max
653                i += 1
654
655            num = 100
656            t = set(np.linspace(y.t_min, t_max, num=num))
657            for i, name in enumerate(scheme.compounds):
658                if not rx.is_transition_state(name):
659                    res = minimize_scalar(
660                        lambda t: -r(t)[i],
661                        bounds=(y.t_min, (t_max + y.t_max) / 2),
662                        method="bounded",
663                    )
664                    if y.t_min < res.x < t_max:
665                        t.update(np.linspace(y.t_min, res.x, num=num // 2))
666                        t.update(np.linspace(res.x, t_max, num=num // 2))
667                        active[i] = True
668
669            t.update(np.geomspace(np.min([_t for _t in t if _t > 0.0]), t_max, num=num))
670            t = np.array(sorted(t))
671            if self.plot not in {"none", None}:
672                if self.plot not in {"all", "active"}:
673                    name = self.plot
674                    plt.plot(t, y(t)[scheme.compounds.index(name)], label=name, lw=3)
675                else:
676                    for i, name in enumerate(scheme.compounds):
677                        if active[i] and not rx.is_transition_state(name):
678                            plt.plot(t, y(t)[i], label=name, lw=3)
679
680                plt.legend()
681                plt.xlabel("Time (s)")
682                plt.ylabel("Concentration (M)")
683                plt.show()

685            if self.savepath is not None:

687                    self.savepath,
688                    np.block([t[:, np.newaxis], y(t).T]),

690                    delimiter=",",

695def _prepare_simulation(scheme, k, concentrations):
696    """
697    Help prepare some data before the simulation.
698
699    Raises
700    ------
701    ValueError
702        If concentrations are invalid or inconsistent with the scheme.
703    """
704    free_y0 = {}
705    fixed_y0 = {}
706    for spec in concentrations:
707        fields = spec.split(":", 1)
708        name, quantity = fields[0].strip(), fields[1].strip()
709
710        if quantity.startswith("!"):
711            d = fixed_y0
712            quantity = quantity[1:]
713        else:
714            d = free_y0
715
716        try:
717            quantity = float(quantity)
718        except (IndexError, ValueError):
719            raise ValueError(
720                "badly formatted concentrations: " f"'{' '.join(concentrations)}'"
721            )
722
723        d[name] = quantity
724
725    # TODO(schneiderfelipe): log stuff related to get_fixed_scheme
726    scheme, k = rx.get_fixed_scheme(scheme, k, fixed_y0)
727
728    y0 = np.zeros(len(scheme.compounds))
729    for compound in free_y0:
730        y0[scheme.compounds.index(compound)] = free_y0[compound]
731
732    return scheme, k, y0
